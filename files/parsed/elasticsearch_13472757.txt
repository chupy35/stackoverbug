It looks like it highlights numbers from your range query. Try replacing the
boolean query in the first "must" clause with this:

    
    
    "filtered": {
        "query": {
            "query_string": {
                "query": "official",
                "default_field": "Content"
            }
        }, 
        "filter": {
            "range": {
                "actualTimeStamp": {
                    "from": "08-04-2011 23:37:06",
                    "to": "16-11-2012 00:00:00"
                }
            }
        }
    }
